\begin{frame}[fragile]
\frametitle{numpy example}

\begin{python}
import numpy as np
A = np.array([
    [0,1,2],
    [2,3,4],
    [4,5,6],
    [6,7,8]])
print A[0,0]
print A[0,1]
print A[1,0]
\end{python}

\note{
    comment on the np abbreviation.
}
\end{frame}

\begin{frame}[fragile]
\frametitle{Slicing}

\begin{python}
import numpy as np
A = np.array([
    [0,1,2],
    [2,3,4],
    [4,5,6],
    [6,7,8]])
print A[0]
print A[1]
print A[:,2]
\end{python}
\end{frame}

\begin{frame}[fragile]
\frametitle{Slices Share Memory!}
\begin{python}
import numpy as np
A = np.array([
    [0,1,2],
    [2,3,4],
    [4,5,6],
    [6,7,8]])
B = A[0]
B[0] = -1
print A[0,0]
\end{python}

\end{frame}
