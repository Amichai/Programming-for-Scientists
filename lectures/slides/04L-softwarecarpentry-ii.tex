\begin{frame}[fragile] % slide 8
\frametitle{}

\begin{python}
def factorial(N):
    '''
    fN = factorial(N)

    Returns the factorial of N.

    N must be equal or greater than zero.
    '''
    if N == 0:
        return 1.
    return N * factorial(N-1)
\end{python}
\end{frame}
